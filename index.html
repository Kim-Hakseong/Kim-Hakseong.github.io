Because We Know the Essence of Programming

Our solutions are built on decades of experience in data acquisition and control systems 

import techsense.core as ts
from datetime import datetime

# Initialize acquisition module
acq = ts.AcquisitionModule(
    sample_rate=1000,  # Hz
    channels=[0, 1, 2, 3],
    precision=24       # bits
)

# Configure real-time processing
def process_data(data):
    return ts.filters.lowpass(
        data, 
        cutoff=50,  # Hz
        order=4
    )

# Start acquisition with callback
acq.start(
    callback=process_data,
    save_path=f"data/{datetime.now().isoformat()}.h5"
)
